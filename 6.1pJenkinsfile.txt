    environment {

        PRODUCTION_ENVIRONMENT = 'Nonefornow'
        EMAIL_RECIPIENTS = 'baldhasmit@gmail.com'

    stages {

            steps {
                script {
                    echo 'Waiting for a short delay...'
                    sleep(time: 1, unit: 'MINUTES')

            steps {

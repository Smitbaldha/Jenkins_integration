pipeline {
    agent any
    
    stages {
        stage('Build') {
            steps {
                script {
                    // Create a log file to capture the build output
                    def logFile = "build-log-${env.BUILD_NUMBER}.txt"
                    // Redirect output to the log file
                    sh "echo 'Building...' > ${logFile}"
                    // Your build steps
                    // Redirect all command outputs to the log file
                    sh "echo 'Executing build steps...' >> ${logFile}"
                    // Example build step
                    sh "echo 'Build completed successfully.' >> ${logFile}"
                }
            }
        }
        stage('Test') {
            steps {
                script {
                    // Append test output to the log file
                    def logFile = "build-log-${env.BUILD_NUMBER}.txt"
                    sh "echo 'Running tests...' >> ${logFile}"
                    // Your test steps
                    sh "echo 'Tests completed successfully.' >> ${logFile}"
                }
            }
        }
    }
    
    post {
        always {
            script {
                // Send the email with the log file attached
                def logFile = "build-log-${env.BUILD_NUMBER}.txt"
                emailext(
                    to: 'recipient@example.com',
                    subject: "Build ${env.JOB_NAME} - ${env.BUILD_NUMBER} Results",
                    body: """Hi Team,
                             
                             The build has completed. Please find the attached build log.
                             
                             Regards,
                             Jenkins""",
                    attachmentsPattern: logFile,  // Attaching the log file
                    recipientProviders: [[$class: 'DevelopersRecipientProvider']]
                )
            }
        }
    }
}



pipeline {
    agent any
    
    stages {
        stage('Build') {
            steps {
                echo 'Building...'
                // Your build steps
            }
        }
        stage('Test') {
            steps {
                echo 'Running tests...'
                // Your test steps
            }
        }
    }
    
    post {
        always {
            script {
                // Write the build log to a file
                def logFile = "build-log-${env.BUILD_NUMBER}.txt"
                sh "cat ${env.WORKSPACE}/log > ${logFile}"

                // Send the email with the log file attached
                emailext(
                    to: 'recipient@example.com',
                    subject: "Build ${env.JOB_NAME} - ${env.BUILD_NUMBER} Results",
                    body: """Hi Team,
                             
                             The build has completed. Please find the attached build log.
                             
                             Regards,
                             Jenkins""",
                    attachmentsPattern: logFile,  // Attaching the log file
                    recipientProviders: [[$class: 'DevelopersRecipientProvider']]
                )
            }
        }
    }
}
